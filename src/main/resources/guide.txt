# Настройки JPA (если вы используете Spring Data JPA)
spring.jpa.hibernate.ddl-auto=update  # Используйте "create", если хотите создавать схему при каждом запуске
spring.jpa.show-sql=true  # Показывать SQL-запросы в консоли
spring.jpa.properties.hibernate.dialect=org.hibernate.dialect.PostgreSQLDialect
